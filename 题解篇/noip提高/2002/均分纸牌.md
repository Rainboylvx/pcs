```c
#include <cstdio>
#include <cstring>

#define N 10010
int n,sum =0,average;
int a[N];

int main(){
    scanf("%d",&n);
    int i,j,k;
    for (i=1;i<=n;i++){
        scanf("%d",&a[i]);
        sum += a[i];
    }

    average = sum / n;

    int cnt =0;
    for (i=1;i<n;i++){
        if( a[i] != average){
            cnt++;
            a[i+1] += (a[i]-average); //借
        }
    }
    printf("%d\n",cnt);
    return 0;
}
```
