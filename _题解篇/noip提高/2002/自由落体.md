#include<iostream>
#include<cstring>
#include<cstdio>
#include<cmath>
using namespace std;
double h,l,k,s1,v,k1,k2;
int n,ans=0;
int main(){
    scanf("%lf %lf %lf %lf %lf %d",&h,&s1,&v,&l,&k,&n);
    k1=s1+l-sqrt((h-k)/5)*v;
    //小球最高时（且求能落在车上）车尾能到的距离
    k2=s1-sqrt(h/5)*v;
    //小球最低时（且求能落在车上）车头能到的距离
    for(int i=0;i<=n-1;i++){
        if((k1-i)>=0.0001&&(k2-i)<=0.0001) ans++;
    }
    printf("%d\n",ans);
    return 0;
}
