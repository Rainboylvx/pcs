```c
#include <iostream>
#include <cmath>
using namespace std;

double dfs(){
    string s;
    cin >> s;
    if( s[0] == '+'){
        return dfs() + dfs();
    }
    else if( s[0] == '-'){
        return dfs() - dfs();
    }
    else if( s[0] == '*'){
        return dfs() * dfs();
    }
    else if( s[0] == '/'){
        return dfs() / dfs();
    }
    return atof(s.c_str());
}

int main(){
    double ans = dfs();
    //std::cout << ans ;
    printf("%f\n", ans);
    return 0;
}
```
