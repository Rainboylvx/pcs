    int i;
    if( d == n+1){
        for(int i=1;i<=n;++i) std::cout << s2[i];
        std::cout << std::endl;
    }
    for(i=1;i<=n;++i){
        if( choose[i] == 1) continue;
        s2[d] = s[i-1];
        choose[i] = 1;
        dfs(d+1);
        choose[i] = 0;
     }
}
